# Convert some data types for some columns
data['Survived'] = data['Survived'].astype('bool')
data['Pclass'] = data['Pclass'].astype('category')
data['Sex'] = data['Sex'].astype('category')
data['Embarked'] = data['Embarked'].astype('category')
data['Age'] = data['Age'].astype('int64')

# Standardize values and ensure correct data types
# standardize Sex to lowercase 'male'/'female' and format Fare to two decimal places.
data['Sex'] = data['Sex'].str.lower()
data['Fare'] = data['Fare'].round(2).astype(float)
display(data)
